#include <stdio.h>

int main() {
    int n, i, j, temp;
    float avg_wait = 0, avg_tat = 0;

    printf("Enter number of processes: ");
    scanf("%d", &n);

    int burst[n], priority[n], process[n], wait[n], tat[n];

    for (i = 0; i < n; i++) {
        printf("Enter burst time for process %d: ", i + 1);
        scanf("%d", &burst[i]);
        printf("Enter priority for process %d: ", i + 1);
        scanf("%d", &priority[i]);
        process[i] = i + 1; // process ID
    }

    
    for (i = 0; i < n - 1; i++) {
        for (j = i + 1; j < n; j++) {
            if (priority[i] > priority[j]) {  
               
                temp = priority[i];
                priority[i] = priority[j];
                priority[j] = temp;

                temp = burst[i];
                burst[i] = burst[j];
                burst[j] = temp;

              
                temp = process[i];
                process[i] = process[j];
                process[j] = temp;
            }
        }
    }

   
    wait[0] = 0;


    for (i = 1; i < n; i++) {
        wait[i] = 0;
        for (j = 0; j < i; j++) {
            wait[i] += burst[j];
        }
    }


    for (i = 0; i < n; i++) {
        tat[i] = burst[i] + wait[i];
    }

    for (i = 0; i < n; i++) {
        avg_wait += wait[i];
        avg_tat += tat[i];
    }

    avg_wait /= n;
    avg_tat /= n;

    printf("\nProcess\tPriority\tBurst Time\tWaiting Time\tTurnaround Time\n");
    printf("---------------------------------------------------------------------\n");
    for (i = 0; i < n; i++) {
        printf("P%d\t\t%d\t\t%d\t\t%d\t\t%d\n", process[i], priority[i], burst[i], wait[i], tat[i]);
    }
    printf("---------------------------------------------------------------------\n");
    printf("Average Waiting Time     = %.2f\n", avg_wait);
    printf("Average Turnaround Time  = %.2f\n", avg_tat);

    return 0;
}
